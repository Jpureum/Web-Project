class Person implements Comparable<Person> {
   int kind;
   String citizenNumber;
   String name;
   int birthYear;
   
   public Person(int kind, String citizenNumber, String name, int birthYear) {
      this.kind = kind;
      this.citizenNumber = citizenNumber;
      this.name = name;
      this.birthYear = birthYear;
   }
   public int getKind() {return kind;}
   public void setKind(int kind) {this.kind = kind;}
   public String getCitizenNumber() {return citizenNumber;}
   public void setCitizenNumber(String citizenNumber) {this.citizenNumber = citizenNumber;}
   public String getName() {return name;}
   public void setName(String name) {this.name = name;}
   public int getBirthYear() {return birthYear;}
   public void setBirthYear(int birthYear) {this.birthYear = birthYear;}
   
   @Override
   public String toString() {
      return citizenNumber+" "+name+" "+String.valueOf(birthYear);
   }
   @Override
   public int compareTo(Person o) {
      return this.citizenNumber.compareTo(o.citizenNumber);
   }
}
